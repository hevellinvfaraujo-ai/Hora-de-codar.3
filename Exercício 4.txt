programa
{
  funcao inicio()
  {
    inteiro soma = 0, quantidade = 0, i
    real media
    
    para (i = 15; i <= 100; i++)
    {
      soma = soma + i
      quantidade++
    }
    
    media = (real)soma / quantidade
    escreva("A média aritmética dos números entre 15 e 100 é: ", media)
  }
}
