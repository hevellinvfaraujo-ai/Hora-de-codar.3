programa
{
  funcao inicio()
  {
    real nota1, nota2, media
    cadeia resposta
    inteiro alunosAprovados = 0
    
    faca
    {
      escreva("Digite a primeira nota: ")
      leia(nota1)
      escreva("Digite a segunda nota: ")
      leia(nota2)
      
      media = (nota1 + nota2) / 2
      escreva("Média final: ", media, "\n")
      
      se (media >= 9.5)
      {
        escreva("APROVADO\n")
        alunosAprovados++
      }
      senao
      {
        escreva("REPROVADO\n")
      }
      
      escreva("Calcular a média de outro aluno? (S/N): ")
      leia(resposta)
    } enquanto (resposta == "S" ou resposta == "s")
    
    escreva("Total de alunos aprovados: ", alunosAprovados)
  }
}
